# Problem: Remove Sub-Folders from the Filesystem - https://leetcode.com/problems/remove-sub-folders-from-the-filesystem/

/**
 * @param {string[]} folder
 * @return {string[]}
 */
var removeSubfolders = function(folder) {
    folder.sort();
    const result=[];
     for (let f of folder) {
        if (!result.length || !f.startsWith(result[result.length - 1] + "/")) {
            result.push(f);
        }
    }
    return result;
};
